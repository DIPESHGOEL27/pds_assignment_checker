#include <stdio.h>
#include <math.h>

int main() {
    float a,u,alpha,t,distance_from_origin;
    printf("Enter the values of a, u, alpha in order :");
    scanf("%f%f%f",&a,&u,&alpha);
    printf("Enter the time :");
    scanf("%f",&t);
    //calculating distance from origin
    distance_from_origin=fabs(a+((u*t)+((0.5)*alpha*t*t)));
    printf("Distance=%f",distance_from_origin);
    return 0;
    
}
